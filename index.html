Business Case

From Revenue Discrepancy to Trusted Single Source of Truth

Revenue data lives across Salesforce (CRM bookings) and SAP BW (billing, revenue recognition). When these systems disagree, Finance loses confidence, executives get inconsistent answers, and planning cycles slow down. Meanwhile, integrations between SAP and third-party systems break silently and errors reach dashboards before anyone catches them. This case demonstrates how I diagnosed the problem, built the data infrastructure to fix it, and delivered a governed reporting layer that the entire Finance organization trusts.

def get_nvidia_fiscal_quarter(dt):
    if isinstance(dt, str):
        dt = datetime.strptime(dt, '%Y-%m-%d')
    month, year = dt.month, dt.year
    if 2 <= month <= 4:
        return f'Q1 FY{year + 1}'
    elif 5 <= month <= 7:
        return f'Q2 FY{year + 1}'
    elif 8 <= month <= 10:
        return f'Q3 FY{year + 1}'
    else:
        fy = year + 1 if month >= 11 else year
        return f'Q4 FY{fy}'

def generate_crm_opportunities(n=500):
    records = []
    for i in range(n):
        cust = random.choice(CUSTOMERS)
        product_sku = random.choice(list(NVIDIA_PRODUCTS.keys()))
        product = NVIDIA_PRODUCTS[product_sku]

        close_date = START_DATE + timedelta(days=random.randint(0, (END_DATE - START_DATE).days))
        create_date = close_date - timedelta(days=random.randint(30, 180))

        if product['category'] == 'Data Center':
            qty = random.choice([8, 16, 32, 64, 128, 256, 512, 1024]) if cust['tier'] in ['Strategic', 'Enterprise'] else random.choice([8, 16, 32, 64])
        elif product['category'] == 'Gaming':
            qty = random.randint(100, 10000)
        elif product['category'] == 'Automotive':
            qty = random.randint(50, 500)
        else:
            qty = random.randint(5, 100)

        unit_price = product['base_price'] + random.uniform(-product['price_var'], product['price_var'])
        tcv = round(qty * unit_price, 2)

        stage_roll = random.random()
        if stage_roll < 0.65:
            stage = 'Closed Won'
        elif stage_roll < 0.80:
            stage = 'Closed Lost'
        elif stage_roll < 0.90:
            stage = 'Negotiation'
        else:
            stage = 'Qualification'

        discount_pct = 0
        if tcv > 5_000_000:
            discount_pct = random.uniform(0.05, 0.20)
        elif tcv > 1_000_000:
            discount_pct = random.uniform(0.02, 0.10)

        net_amount = round(tcv * (1 - discount_pct), 2)

        records.append({
            'opportunity_id': f'OPP-{10000 + i}',
            'sfdc_account_id': cust['sfdc_id'],
            'account_name': cust['name'],
            'region': cust['region'],
            'customer_segment': cust['segment'],
            'product_sku': product_sku,
            'product_category': product['category'],
            'quantity': qty,
            'unit_price_usd': round(unit_price, 2),
            'total_contract_value_usd': tcv,
            'discount_pct': round(discount_pct, 4),
            'net_booking_amount_usd': net_amount,
            'stage': stage,
            'close_date': close_date.strftime('%Y-%m-%d'),
            'created_date': create_date.strftime('%Y-%m-%d'),
            'fiscal_quarter': get_nvidia_fiscal_quarter(close_date),
            'sales_rep': random.choice(SALES_REPS),
            'deal_type': random.choice(['New', 'Expansion', 'Renewal']),
        })
    return pd.DataFrame(records)

def generate_sap_billing(crm_df):
    closed_won = crm_df[crm_df['stage'] == 'Closed Won'].copy()
    billing_records = []
    discrepancy_log = []

    for _, deal in closed_won.iterrows():
        cust_match = [c for c in CUSTOMERS if c['sfdc_id'] == deal['sfdc_account_id']]
        sap_id = cust_match[0]['sap_id'] if cust_match and cust_match[0]['sap_id'] else None

        if sap_id is None:
            discrepancy_log.append({
                'opportunity_id': deal['opportunity_id'],
                'type': 'MISSING_SAP_MAPPING',
                'detail': f"No SAP customer ID for SFDC {deal['sfdc_account_id']}"
            })
            continue

        close_dt = datetime.strptime(deal['close_date'], '%Y-%m-%d')

        # 12% timing differences
        if random.random() < 0.12:
            billing_date = close_dt + timedelta(days=random.randint(5, 45))
            discrepancy_log.append({
                'opportunity_id': deal['opportunity_id'],
                'type': 'TIMING_DIFFERENCE',
                'detail': f"Billing {(billing_date - close_dt).days} days after CRM close"
            })
        else:
            billing_date = close_dt + timedelta(days=random.randint(0, 3))

        # 8% split billing on large deals
        if deal['net_booking_amount_usd'] > 2_000_000 and random.random() < 0.08:
            splits = random.choice([2, 3])
            amounts = np.random.dirichlet(np.ones(splits)) * deal['net_booking_amount_usd']
            for s in range(splits):
                bill_date = billing_date + timedelta(days=s * random.randint(15, 45))
                billing_records.append({
                    'billing_doc_id': f'SAP-BD-{random.randint(100000, 999999)}',
                    'sap_customer_id': sap_id,
                    'sales_order_ref': deal['opportunity_id'],
                    'product_sku': deal['product_sku'],
                    'billing_date': bill_date.strftime('%Y-%m-%d'),
                    'fiscal_period': get_nvidia_fiscal_quarter(bill_date),
                    'billed_quantity': deal['quantity'] // splits,
                    'billed_amount_usd': round(amounts[s], 2),
                    'revenue_type': 'PRODUCT' if s == 0 else 'DEFERRED',
                    'posting_status': 'POSTED',
                    'plant_code': random.choice(['P100', 'P200', 'P300']),
                    'company_code': '1000',
                })
            discrepancy_log.append({
                'opportunity_id': deal['opportunity_id'],
                'type': 'SPLIT_BILLING',
                'detail': f"Deal split into {splits} billing documents"
            })
            continue

        # 5% amount variances
        amount_variance = 0
        if random.random() < 0.05:
            amount_variance = random.uniform(-500, 500)
            discrepancy_log.append({
                'opportunity_id': deal['opportunity_id'],
                'type': 'AMOUNT_VARIANCE',
                'detail': f"SAP amount differs by ${amount_variance:.2f}"
            })

        # 3% unbilled backlog
        if random.random() < 0.03:
            discrepancy_log.append({
                'opportunity_id': deal['opportunity_id'],
                'type': 'UNBILLED',
                'detail': 'Deal closed in CRM but not yet billed in SAP'
            })
            continue

        billing_records.append({
            'billing_doc_id': f'SAP-BD-{random.randint(100000, 999999)}',
            'sap_customer_id': sap_id,
            'sales_order_ref': deal['opportunity_id'],
            'product_sku': deal['product_sku'],
            'billing_date': billing_date.strftime('%Y-%m-%d'),
            'fiscal_period': get_nvidia_fiscal_quarter(billing_date),
            'billed_quantity': deal['quantity'],
            'billed_amount_usd': round(deal['net_booking_amount_usd'] + amount_variance, 2),
            'revenue_type': 'PRODUCT',
            'posting_status': 'POSTED',
            'plant_code': random.choice(['P100', 'P200', 'P300']),
            'company_code': '1000',
        })

    return pd.DataFrame(billing_records), pd.DataFrame(discrepancy_log)

def generate_customer_master():
    records = []
    for c in CUSTOMERS:
        records.append({
            'sfdc_account_id': c['sfdc_id'],
            'sap_customer_id': c['sap_id'] if c['sap_id'] else 'UNMAPPED',
            'customer_name': c['name'],
            'region': c['region'],
            'customer_segment': c['segment'],
            'tier': c['tier'],
            'mapping_status': 'ACTIVE' if c['sap_id'] else 'PENDING_MAPPING',
            'last_validated': '2025-12-15' if c['sap_id'] else None,
            'data_steward': random.choice(['Finance Ops', 'Sales Ops', 'Revenue Accounting']),
        })
    return pd.DataFrame(records)

02_data_quality_checks.sql 
-- Bookings with invalid customer keys
SELECT 'REFERENTIAL_INTEGRITY' AS check_type,
       'fact_crm_bookings.customer_key' AS check_target,
       COUNT(*) AS error_count,
       CASE WHEN COUNT(*) = 0 THEN 'PASS' ELSE 'FAIL' END AS status
FROM fact_crm_bookings b
LEFT JOIN dim_customer c ON b.customer_key = c.customer_key
WHERE c.customer_key IS NULL;

-- Billing docs with invalid customer keys
SELECT 'REFERENTIAL_INTEGRITY' AS check_type,
       'fact_sap_billing.customer_key' AS check_target,
       COUNT(*) AS error_count,
       CASE WHEN COUNT(*) = 0 THEN 'PASS' ELSE 'FAIL' END AS status
FROM fact_sap_billing b
LEFT JOIN dim_customer c ON b.customer_key = c.customer_key
WHERE c.customer_key IS NULL;

-- CRM null rates on critical fields
SELECT 'NULL_RATE' AS check_type,
       'fact_crm_bookings.net_booking_amount' AS check_target,
       ROUND(SUM(CASE WHEN net_booking_amount IS NULL THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 2) AS null_pct,
       CASE WHEN SUM(CASE WHEN net_booking_amount IS NULL THEN 1 ELSE 0 END) * 100.0 / COUNT(*) > 1.0
            THEN 'FAIL' ELSE 'PASS' END AS status
FROM fact_crm_bookings;

-- SAP customer ID mapping gaps
SELECT 'NULL_RATE' AS check_type,
       'dim_customer.sap_customer_id' AS check_target,
       ROUND(SUM(CASE WHEN sap_customer_id IS NULL OR sap_customer_id = 'UNMAPPED' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 2) AS null_pct,
       CASE WHEN SUM(CASE WHEN sap_customer_id IS NULL OR sap_customer_id = 'UNMAPPED' THEN 1 ELSE 0 END) * 100.0 / COUNT(*) > 5.0
            THEN 'WARN' ELSE 'PASS' END AS status
FROM dim_customer;

-- Row count variance vs 7-day rolling average
WITH daily_counts AS (
    SELECT DATE(loaded_at) AS load_date, COUNT(*) AS row_count
    FROM fact_crm_bookings
    GROUP BY DATE(loaded_at)
),
rolling_avg AS (
    SELECT load_date, row_count,
           AVG(row_count) OVER (ORDER BY load_date ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING) AS avg_7d
    FROM daily_counts
)
SELECT 'ROW_COUNT_VARIANCE' AS check_type,
       load_date AS check_target,
       row_count, ROUND(avg_7d, 1) AS rolling_avg_7d,
       ROUND((row_count - avg_7d) / NULLIF(avg_7d, 0) * 100, 1) AS variance_pct,
       CASE WHEN ABS((row_count - avg_7d) / NULLIF(avg_7d, 0)) > 0.5 THEN 'FAIL'
            WHEN ABS((row_count - avg_7d) / NULLIF(avg_7d, 0)) > 0.25 THEN 'WARN'
            ELSE 'PASS' END AS status
FROM rolling_avg
WHERE load_date = CURRENT_DATE;

-- Negative amounts
SELECT 'VALUE_RANGE' AS check_type,
       'fact_crm_bookings.net_booking_amount' AS check_target,
       COUNT(*) AS out_of_range_count,
       CASE WHEN COUNT(*) > 0 THEN 'WARN' ELSE 'PASS' END AS status
FROM fact_crm_bookings
WHERE net_booking_amount < 0 OR net_booking_amount > 500000000;

-- CRM-to-SAP reconciliation by customer and period
WITH crm_totals AS (
    SELECT c.sfdc_account_id, c.customer_name, b.time_key,
           SUM(b.net_booking_amount) AS crm_total
    FROM fact_crm_bookings b
    JOIN dim_customer c ON b.customer_key = c.customer_key
    WHERE b.stage = 'Closed Won'
    GROUP BY c.sfdc_account_id, c.customer_name, b.time_key
),
sap_totals AS (
    SELECT c.sfdc_account_id, s.time_key,
           SUM(s.billed_amount_usd) AS sap_total
    FROM fact_sap_billing s
    JOIN dim_customer c ON s.customer_key = c.customer_key
    GROUP BY c.sfdc_account_id, s.time_key
)
SELECT 'RECONCILIATION' AS check_type,
       crm.sfdc_account_id AS check_target,
       crm.crm_total,
       COALESCE(sap.sap_total, 0) AS sap_total,
       crm.crm_total - COALESCE(sap.sap_total, 0) AS variance,
       CASE WHEN ABS(crm.crm_total - COALESCE(sap.sap_total, 0)) / NULLIF(crm.crm_total, 0) > 0.05 THEN 'FAIL'
            WHEN ABS(crm.crm_total - COALESCE(sap.sap_total, 0)) / NULLIF(crm.crm_total, 0) > 0.01 THEN 'WARN'
            ELSE 'PASS' END AS status
FROM crm_totals crm
LEFT JOIN sap_totals sap ON crm.sfdc_account_id = sap.sfdc_account_id AND crm.time_key = sap.time_key
ORDER BY ABS(crm.crm_total - COALESCE(sap.sap_total, 0)) DESC;

-- Duplicate detection
SELECT 'DUPLICATE' AS check_type,
       'fact_crm_bookings.opportunity_id' AS check_target,
       COUNT(*) - COUNT(DISTINCT opportunity_id) AS duplicate_count,
       CASE WHEN COUNT(*) > COUNT(DISTINCT opportunity_id) THEN 'FAIL' ELSE 'PASS' END AS status
FROM fact_crm_bookings;

-- Data freshness
SELECT 'FRESHNESS' AS check_type,
       'fact_crm_bookings' AS check_target,
       MAX(loaded_at) AS last_loaded,
       TIMESTAMPDIFF(HOUR, MAX(loaded_at), NOW()) AS hours_since_load,
       CASE WHEN TIMESTAMPDIFF(HOUR, MAX(loaded_at), NOW()) > 24 THEN 'FAIL'
            WHEN TIMESTAMPDIFF(HOUR, MAX(loaded_at), NOW()) > 12 THEN 'WARN'
            ELSE 'PASS' END AS status
FROM fact_crm_bookings;

import pandas as pd
import numpy as np
from datetime import datetime
import json
import os

class ReconciliationEngine:
    THRESHOLDS = {
        'amount_variance_pct': 0.01,
        'amount_variance_abs': 500,
        'timing_days': 5,
        'null_rate_critical': 0.02,
        'reconciliation_target': 0.95,
    }

    def __init__(self, crm_path, sap_path, customer_master_path):
        self.crm_df = pd.read_csv(crm_path)
        self.sap_df = pd.read_csv(sap_path)
        self.customer_master = pd.read_csv(customer_master_path)
        self.results = {}
        self.alerts = []
        self.run_timestamp = datetime.now().isoformat()

    def run_full_reconciliation(self):
        self._check_data_quality()
        self._reconcile_customer_master()
        self._reconcile_revenue()
        self._generate_summary()
        return self.results

    def _check_data_quality(self):
        checks = []
        for col in ['opportunity_id', 'sfdc_account_id', 'net_booking_amount_usd', 'close_date']:
            null_rate = self.crm_df[col].isna().mean()
            status = 'FAIL' if null_rate > self.THRESHOLDS['null_rate_critical'] else 'PASS'
            checks.append({'table': 'CRM', 'column': col, 'null_rate': round(null_rate, 4), 'status': status})
            if status == 'FAIL':
                self.alerts.append(f"CRITICAL: CRM column '{col}' has {null_rate:.1%} null rate")

        for col in ['billing_doc_id', 'sap_customer_id', 'billed_amount_usd']:
            null_rate = self.sap_df[col].isna().mean()
            status = 'FAIL' if null_rate > self.THRESHOLDS['null_rate_critical'] else 'PASS'
            checks.append({'table': 'SAP', 'column': col, 'null_rate': round(null_rate, 4), 'status': status})

        crm_dupes = self.crm_df['opportunity_id'].duplicated().sum()
        sap_dupes = self.sap_df['billing_doc_id'].duplicated().sum()
        checks.append({'table': 'CRM', 'column': 'opportunity_id', 'duplicates': crm_dupes, 'status': 'FAIL' if crm_dupes > 0 else 'PASS'})
        checks.append({'table': 'SAP', 'column': 'billing_doc_id', 'duplicates': sap_dupes, 'status': 'FAIL' if sap_dupes > 0 else 'PASS'})

        self.results['data_quality'] = pd.DataFrame(checks)

    def _reconcile_customer_master(self):
        master = self.customer_master.copy()
        unmapped = master[master['mapping_status'] == 'PENDING_MAPPING']
        active = master[master['mapping_status'] == 'ACTIVE']

        crm_accounts = set(self.crm_df['sfdc_account_id'].unique())
        master_accounts = set(master['sfdc_account_id'].unique())
        orphan_crm = crm_accounts - master_accounts

        self.results['customer_master'] = {
            'total_customers': len(master),
            'active_mappings': len(active),
            'pending_mappings': len(unmapped),
            'orphan_crm_accounts': len(orphan_crm),
            'mapping_coverage': round(len(active) / max(len(master), 1), 4),
        }

        if len(unmapped) > 0:
            self.alerts.append(f"WARNING: {len(unmapped)} customers have no SAP mapping")

    def _reconcile_revenue(self):
        closed_won = self.crm_df[self.crm_df['stage'] == 'Closed Won'].copy()

        sap_by_opp = self.sap_df.groupby('sales_order_ref').agg(
            sap_total=('billed_amount_usd', 'sum'),
            billing_count=('billing_doc_id', 'count'),
            first_billing_date=('billing_date', 'min'),
        ).reset_index()

        merged = closed_won.merge(
            sap_by_opp, left_on='opportunity_id', right_on='sales_order_ref', how='left'
        )

        merged['sap_total'] = merged['sap_total'].fillna(0)
        merged['variance_amount'] = merged['net_booking_amount_usd'] - merged['sap_total']
        merged['variance_pct'] = merged['variance_amount'] / merged['net_booking_amount_usd'].replace(0, np.nan)

        conditions = [
            merged['sap_total'] == 0,
            merged['variance_pct'].abs() > self.THRESHOLDS['amount_variance_pct'],
            merged['variance_amount'].abs() > self.THRESHOLDS['amount_variance_abs'],
        ]
        labels = ['UNBILLED', 'PCT_VARIANCE', 'ABS_VARIANCE']

        merged['discrepancy_type'] = 'MATCHED'
        for cond, label in zip(conditions, labels):
            merged.loc[cond & (merged['discrepancy_type'] == 'MATCHED'), 'discrepancy_type'] = label

        total = len(merged)
        matched = len(merged[merged['discrepancy_type'] == 'MATCHED'])
        match_rate = matched / max(total, 1)

        self.results['revenue_reconciliation'] = {
            'total_closed_won': total,
            'matched': matched,
            'match_rate': round(match_rate, 4),
            'total_crm_amount': round(merged['net_booking_amount_usd'].sum(), 2),
            'total_sap_amount': round(merged['sap_total'].sum(), 2),
            'total_variance': round(merged['variance_amount'].sum(), 2),
            'discrepancy_breakdown': merged['discrepancy_type'].value_counts().to_dict(),
        }

        self.results['reconciliation_detail'] = merged

        if match_rate < self.THRESHOLDS['reconciliation_target']:
            self.alerts.append(f"CRITICAL: Match rate {match_rate:.1%} below 95% target")

    def _generate_summary(self):
        recon = self.results['revenue_reconciliation']
        summary = {
            'run_timestamp': self.run_timestamp,
            'dq_checks_passed': int((self.results['data_quality']['status'] == 'PASS').sum()),
            'customer_mapping_coverage': self.results['customer_master']['mapping_coverage'],
            'reconciliation_match_rate': recon['match_rate'],
            'total_crm_revenue': recon['total_crm_amount'],
            'total_sap_revenue': recon['total_sap_amount'],
            'net_variance': recon['total_variance'],
            'alerts': self.alerts,
        }
        self.results['executive_summary'] = summary
        return summary

    def export_reports(self, output_dir):
        with open(os.path.join(output_dir, 'reconciliation_summary.json'), 'w') as f:
            json.dump(self.results['executive_summary'], f, indent=2, default=str)
        if 'reconciliation_detail' in self.results:
            self.results['reconciliation_detail'].to_csv(
                os.path.join(output_dir, 'reconciliation_detail.csv'), index=False
            )

if __name__ == '__main__':
    base_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
    data_dir = os.path.join(base_dir, 'data')
    engine = ReconciliationEngine(
        crm_path=os.path.join(data_dir, 'crm_opportunities.csv'),
        sap_path=os.path.join(data_dir, 'sap_billing.csv'),
        customer_master_path=os.path.join(data_dir, 'customer_master.csv'),
    )
    engine.run_full_reconciliation()
    engine.export_reports(data_dir)
